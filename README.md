## Install

```bash
npm install --save react-axios-swagger-gen
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'react-axios-swagger-gen'
import 'react-axios-swagger-gen/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
